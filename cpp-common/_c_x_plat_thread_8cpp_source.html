00187 CXPlatThread::THREAD_STATUS CXPlatThread::GetThreadStatus()
00188 {
00189     // Not even allocated?
00190     if (!hThreadHandle)
00191         return (NOT_STARTED);
00192 
00193 #if !defined(_WIN32)
00194     return (IsThreadActive(hThreadHandle) ? RUNNING : THREAD_TERMINATED);
00195 #else
00196     DWORD dwTerminationStatus;
00197 
00198     // Will fail if not a valid handle
00199     if (!::GetExitCodeThread(hThreadHandle, (unsigned long *) &dwTerminationStatus))
00200         return (THREAD_TERMINATED);
00201     //Check termination status
00202     if (dwTerminationStatus == STILL_ACTIVE)
00203         return (bSuspended ? SUSPENDED : RUNNING);
00204     return (THREAD_TERMINATED);

00209 // Wait for a thread to terminate.  After timeout, kill it with extreme prejudice.
00210 
00211 DWORD CXPlatThread::WaitForThreadCompletion(DWORD dwTimeout)
00212 {
00213 #if defined(_WIN32)     
00214     if (!hThreadHandle)
00215         return 0;
00216     else
00217         return (::WaitForSingleObject(hThreadHandle, dwTimeout));
00218 #else
00219     if (!hThreadHandle)
00220         return (0);
00221     if (bDetached)
00222         return (INVALID_REQUEST);
00223     if (bJoined)
00224         return (0);
00225     // Bogus kudge because pthread_join doesn't have a timeout
00226     if (dwTimeout != INFINITE)
00227     {
00228         int iMaxWait = dwTimeout;
00229         int iCurWait = 0;
00230         BOOL bTimeout = TRUE;
00231 
00232         do
00233         { // Wait for the thread to complete
00234             if (!IsThreadActive(hThreadHandle))
00235             { // Thread completed
00236                 bTimeout = FALSE;
00237                 break;
00238             };
00239             usleep(1); // Wait timeout for it to complete
00240         }
00241         while(++iCurWait < iMaxWait);
00242         if (bTimeout)
00243             ::pthread_cancel(hThreadHandle);
00244     }
00245 
00246     bJoined = TRUE;
00247     int iRet = 0;
00248     if (hThreadHandle)
00249     {
00250         iRet = ::pthread_join(hThreadHandle, NULL);
00251         ::memset(&hThreadHandle, 0, sizeof(hThreadHandle));
00252     }
00253     return (iRet);
00254 #endif
00255 };
