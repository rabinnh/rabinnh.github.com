00021 //
00022 //      Created: Richard A. Bross
00023 //
00025 //              These threads are used and managed by the CThreadPool
00026 //      class.  You should derive a class from this class
00027 //      and implement a PooledWorkerFunction to do your work.
00028 //
00029 //      The PooledWorkerFunction should do a job and return,
00030 //      allowing this thread to be reused.
00031 //
00032 //      The thread is created on construction, but work will
00033 //      not begin until you call "StartWorking".  There is
00034 //      not an "EndWorking" call, because if the thread is
00035 //      busy, it is in the user's code.  The thread is
00036 //      destroyed on object destruction.
00037 //
00038 //      You can check the IsThreadBusy method to see if
00039 //      the thread is busy or if it is available for work.
00040 //
00041 //      If a CXPlatEvent has been passed into
00042 //      the constructor, the thread will set it when it is
00043 //      available for work.
00044

00086 // Implemented by this class. SHOULD NOT BE DERIVED FROM!
00087 // Use PooledWorkerFunction instead.
00088 
00089 unsigned CXPlatPooledThread::WorkerFunction(void *pParam)
00090 {
00091     pFuncParam = pParam;
00092     unsigned uReturn = 0;
00093     do
00094     {   // Wait for the "Start working" signal
00095         cStartWorking.Lock();
00096         // Check the terminate flag
00097         if (bTerminateThread)
00098             return(uReturn);
00099         // We're busy
00100         bThreadBusy = TRUE;
00101 
00102         // Call the user's function
00103         uReturn = PooledWorkerFunction();
00104         // They're done.  We're no longer busy.
00105         bThreadBusy = bThreadAllocated = FALSE;
00106         // Notify the thread pool that we are once again available.
00107         if (pPoolParent)
00108             pPoolParent->cThreadAvailableEvent.SetEvent();
00109     }
00110     while(!bTerminateThread);
00111 
00112     return(uReturn);
00113 };
