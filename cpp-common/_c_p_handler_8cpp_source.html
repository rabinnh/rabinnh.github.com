00045         // Add a connection to the queued.
00046         // These are added to cQueuedConnections.
00047         // At the beginning of the next I/O cycle, they are added to cLiveConnections.
00048 BOOL CPHandler::AddConnection(CPConnection *pConnection, SOCKET sNewSocket)
00049 {
00050         CONNECTION_ITERATOR     IterSocket;
00051         unsigned char           *ucHS;
00052         int                                     nReturn;
00053 
00054                 // Check pointer
00055         if (!pConnection)
00056                 return(FALSE);
00057 
00058                 // If not connected, forget it
00059         if (pConnection->eConnected != CPConnection::CONNECT_SUCCESS
00060                                                 && pConnection->eConnected != CPConnection::CONNECT_PENDING)
00061                 return(FALSE);
00062 
00063         cCritQueuedMap.Lock();
00064 
00065                 // Assign the socket
00066         if (sNewSocket != INVALID_SOCKET)
00067                 {
00068                 pConnection->sSocket = sNewSocket;
00069                 pConnection->SetBlockingMode(FALSE);
00070                 };
00071 
00072                 // Timestamp.
00073         pConnection->tConnected = time(NULL);
00074 
00075         // See if it's already there
00076     IterSocket = cQueuedConnections.find(pConnection->cSignature);
00077 
00078         // Found?
00079     if (IterSocket != cQueuedConnections.end())
00080                 {       // This socket is already in the map
00081                 cCritQueuedMap.Unlock();
00082                 return(TRUE);   // Although technically an error, for calling it twice, it IS in the map.
00083                 }
00084         else
00085                 {       // Add
00086                 cQueuedConnections.insert(CONNECTION_MAP::value_type(pConnection->cSignature, pConnection));
00087                 pConnection->SetHandler(this);
00088                 if (pConnection->GetHandshake(&ucHS, nReturn))
00089                         pConnection->SendDataOut((char *) ucHS, 4096, nReturn);
00090                 iSize++;
00091                 };
00092 
00093         cCritQueuedMap.Unlock();
00094 
00095         return(TRUE);
00096 };

00100         // Remove a connection from cLiveConnections.
00101         // This will wait until the next I/O cycle is complete.
00102         // This is usually only called from a CPConnection
00103         // destructor, since it has to wait to be removed before
00104         // being deleted.
00105 BOOL CPHandler::RemoveConnection(CPConnection *pConnection)
00106 {
00107         CONNECTION_ITERATOR     IterSocket;
00108         CONNECTION_ITERATOR     IterSocket2;
00109 
00110                 // If it's not connected, forget it.
00111         if (!pConnection)
00112                 return(FALSE);
00113 
00114                 // First see if it's in the queue map
00115         cCritQueuedMap.Lock();
00116 
00117         // See if it's already there
00118     IterSocket = cQueuedConnections.find(pConnection->cSignature);
00119         // Found?
00120     if (IterSocket != cQueuedConnections.end())
00121                 {
00122                 IterSocket2 = IterSocket;       
00123                 IterSocket2++;
00124                 cQueuedConnections.erase(IterSocket);   
00125                 IterSocket = IterSocket2;
00126                 pConnection->SetHandler(NULL);
00127                 iSize--;
00128                 }
00129 
00130         cCritQueuedMap.Unlock();
00131 
00132         cCritMap.Lock();
00133 
00134         // See if it's already there
00135     IterSocket = cLiveConnections.find(pConnection->cSignature);
00136         // Found?
00137     if (IterSocket != cLiveConnections.end())
00138                 {
00139                 IterSocket2 = IterSocket;       
00140                 IterSocket2++;
00141                 cLiveConnections.erase(IterSocket);     
00142                 IterSocket = IterSocket2;
00143                 pConnection->SetHandler(NULL);
00144                 iSize--;
00145                 }
00146         else
00147                 {       // Not found, which was the point, I guess.
00148                 }
00149 
00150         cCritMap.Unlock();
