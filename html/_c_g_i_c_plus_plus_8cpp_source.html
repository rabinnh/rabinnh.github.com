00045 bool CGICPlusPlus::ParseRequest()
00046 {
00047     register int i;
00048     char *request_method;
00049     int content_length;
00050     char *cgiinput;
00051     char *nvpair;
00052     char *eqpos;
00053 
00054     // Depending on the request method, read all CGI input into cgiinput.
00055     request_method = getenv("REQUEST_METHOD");
00056 
00057     if (!strcmp(request_method, "GET") || !strcmp(request_method, "HEAD"))
00058     {
00059         char *qs;
00060         qs = getenv("QUERY_STRING");
00061         cgiinput = strdup(qs ? qs : "");
00062     }
00063     else if (!strcmp(request_method, "POST") || !strcmp(request_method, "PUT"))
00064     {
00065         if (strncasecmp(getenv("CONTENT_TYPE"), CONTENT_FORM, ::strlen(CONTENT_FORM)))
00066             return false;
00067         if (!(content_length = atoi(getenv("CONTENT_LENGTH"))))
00068             return false;
00069         // Read the input data
00070         if (!(cgiinput = (char *) malloc(content_length + 1)))
00071             return false;
00072         if (!fread(cgiinput, content_length, 1, stdin))
00073             return false;
00074         cgiinput[content_length] = '\0';
00075     }
00076     else
00077         return false;
00078 
00079     // Change all plusses back to spaces.
00080     for (i = 0; cgiinput[i]; i++)
00081         if (cgiinput[i] == '+')
00082             cgiinput[i] = ' ';
00083 
00084     // Split on "&" and ";" to extract the name-value pairs into
00085     nvpair = strtok(cgiinput, "&;");
00086     while (nvpair)
00087     {
00088         string sKey;
00089         string sValue;
00090         // Get the value
00091         if (eqpos = strchr(nvpair, '='))
00092         {
00093             *eqpos = '\0';
00094             unescape_url(eqpos + 1);
00095         }
00096         else
00097         { // If no '=' sign, set the value to blank
00098             *(eqpos + 1) = '\0';
00099         }
00100         // Now escape the key
00101         unescape_url(nvpair);
00102         // Set the strings
00103         sKey = nvpair;
00104         sValue = eqpos + 1;
00105         // Save in the mape
00106         mCgiMap[sKey] = sValue;
00107         // Get the next one
00108         nvpair = strtok(NULL, "&;");
00109     }
00110 
00111     // Free buffer
00112     free(cgiinput);
00113 
00114     // We were successful
00115     return true;
00116 }

00198 // Convert a two-char hex string into a char
00199 
00200 char CGICPlusPlus::x2c(char *what)
00201 {
00202     register char digit;
00203 
00204     digit = (what[0] >= 'A' ? ((what[0] & 0xdf) - 'A') + 10 : (what[0] - '0'));
00205     digit *= 16;
00206     digit += (what[1] >= 'A' ? ((what[1] & 0xdf) - 'A') + 10 : (what[1] - '0'));
00207 
00208     return (digit);
