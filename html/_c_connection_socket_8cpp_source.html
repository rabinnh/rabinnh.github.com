00051 // Requests that the handleconenction thread close
00052 // NOTES:
00053 //  We used to just close the socket.  However, data waiting in the socket buffer was lost.
00054 //  We still want the thread to exit on an invalid socket, since the other side may close the connection.
00055 //  Now we set a flag and let the HandleConnection thread get remaining data.
00056 //  If a user is calling this and they are stupid enough to call SendDataOut at the same time,
00057 //  tough luck, the connection will be closed after 50 ms max.

00059 void CConnectionSocket::TerminateConnection(ECLOSE_TYPE eRequest)
00060 {
00061     // Ask the connection thread to close
00062     CProfileTimer::SetTimestamp("CCS-CLOSE");
00063     eRequestClose = eRequest;
00064     if (hThread)
00065     {
00066 #if defined(_WIN32)
00067         ::WaitForSingleObject(hThread, INFINITE);
00068         ::CloseHandle(hThread); // _beginthreadex doesn't auto close the handle
00069 #else
00070         // Make sure thread finishes
00071         int iRet;
00072         if ((iRet = ::pthread_join(hThread, NULL)) != 0)
00073         { // EDEADLK (35) will occur if some bozo calls this routine from the thread itself
00074             ::openlog("connectionsocket", LOG_PID, LOG_LOCAL0);
00075             char strLogEntry[1024];
00076             ::sprintf(strLogEntry, "pthread_join error could cause memory leak.  Error number: %d", iRet);
00077             ::syslog(LOG_CRIT, "%s", strLogEntry);
00078             ::closelog();
00079         }
00080         else
00081         { // Only on successs
00082             ::memset(&hThread, 0, sizeof(hThread));
00083         }
00084 
00085 #endif
00086     };
00087     // Close the socket
00088     CloseSocket();
00089 };
