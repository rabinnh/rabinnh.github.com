00089 // Get difference between timestamps
00090 
00091 ElapsedTime CProfileTimer::GetDifference(ElapsedTime &cElapsed, struct timeval *sT1, struct timeval *sT2)
00092 {
00093     cElapsed.lMS = 0;
00094     cElapsed.lSec = 0;
00095     cElapsed.lUS = 0;
00096     
00097     // First convert to ms, usec to make it easier
00098     // This can't be negative, just >= 0
00099     cElapsed.lMS = (sT2->tv_sec - sT1->tv_sec) * 1000;  // Sec to ms
00100     cElapsed.lUS = sT2->tv_usec - sT1->tv_usec; // usec
00101     if (cElapsed.lUS < 0)
00102     {   // This is based on millions, so
00103         cElapsed.lUS = 1000000 + cElapsed.lUS;
00104         cElapsed.lMS -= 1000; // And that's one less second
00105     }
00106     // Make it secs:ms:usec
00107     cElapsed.lMS += cElapsed.lUS / 1000;    // How many ms are in the usec field?
00108     cElapsed.lUS %= 1000;                   // Remainder
00109     cElapsed.lSec = cElapsed.lMS / 1000;    // How many secs are in the ms field?
00110     cElapsed.lMS %= 1000;                   // Remainder
00111 }
