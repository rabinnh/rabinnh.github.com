00088

00104         if (bWaitToFinishSet)
00105             pThread->SetCompletionWaitTime(lWaitToFinish);

00138 int CXPlatThreadPool::GetAvailableThreads(THREAD_POOL_ARRAY &cAvailThreadPool, int iNeeded, BOOL bWaitForAtLeastOne, DWORD dwWaitTime)
00139 {
00140     int iIndex;
00141     int iSize;
00142     CXPlatPooledThread *pThread;
00143     BOOL bSuccess = FALSE;
00144     int iReturn = 0;
00145 
00146     // Clear the passed in array
00147     cAvailThreadPool.clear();
00148 
00149     // Make sure the call is valid
00150     if (!iNeeded)
00151         return (TP_ZERO_THREADS_REQUESTED);
00152 
00153     // Make sure there are enough.
00154     // If even one won't do and they are asking
00155     // for more than are allowed, it's an error.
00156     if (!bWaitForAtLeastOne && iNeeded > iMaxThreadsAllowed)
00157         return (TP_REQUEST_EXCEEDS_POOL_SIZE);
00158 
00159     // Lock access to the thread pool
00160     cPoolCritSect.Lock();
00161 
00162     // Meet the demand
00163     do
00164     {   
00165         bool bShrunk = false;
00166         
00167         // Clear the passed in array to start over.
00168         cAvailThreadPool.clear();
00169         iSize = cThreadPool.size();
00170         // Load their array with the available thread objects
00171         for (iIndex = 0; iIndex < iSize; iIndex++)
00172         {
00173             pThread = cThreadPool[iIndex];
00174             if (!pThread)
00175                 continue; // How could this ever happen? Defensive coding.
00176 
00177             // We don't have to worry about a race condition with
00178             // bThreadAllocated because it's only checked and set TRUE here,
00179             // and we're protected by the critical section.
00180             if (pThread->bThreadAllocated)
00181                 continue;
00182             
00183             // See if we need to shrink the pool
00184             if (iSize > iMaxThreadsAllowed) 
00185             {
00186                 delete pThread;
00187                 cThreadPool.erase(cThreadPool.begin() + iIndex);
00188                 bShrunk = true;
00189                 break;
00190             }
00191             
00192             cAvailThreadPool.push_back(pThread);
00193             if (cAvailThreadPool.size() == iNeeded)
00194                 break;
00195         };
00196         
00197         // If we shrunk the pool, try again
00198         if (bShrunk)
00199             continue;
00200 
00201         // If we've gotten all we needed and there was no shrinkage, cool.
00202         if (cAvailThreadPool.size() == iNeeded)
00203         {
00204             bSuccess = TRUE;
00205             break;
00206         }
00207 
00208         // We didn't get all the threads we needed.
00209         // First, see if there's room to grow
00210         if (GrowThreadPool())
00211             continue; // We grew the thread pool, so try again.
00212 
00213         // Did they want to wait for at least one?
00214         if (!bWaitForAtLeastOne) // They wanted all or none.
00215         {
00216             iReturn = TP_NOT_ENOUGH_THREADS_AVAILABLE;
00217             break;
00218         };
00219 
00220         // Couldn't grow the pool.  Is at least one sufficient?
00221         if (cAvailThreadPool.size())
00222         {
00223             bSuccess = TRUE;
00224             break;
00225         };
00226 
00227         // Wait for a thread to become available.
00228         // In *nix systems you have to be careful.  Unlike Windows, when an object is set before 
00229         // pthread_cond_wait, it does not recognize that it is signaled.  This is an issue with multiple
00230         // threads setting the event.  So in that rare case, check every 20 ms just to be sure
00231         iReturn = cThreadAvailableEvent.Lock(dwWaitTime);
00232         if (iReturn == XPLAT_TIMEOUT)
00233         {
00234             iReturn = TP_WAIT_TIMED_OUT;
00235             break;
00236         };
00237         // At least one came free.  Start over.
00238     }
00239     while (!bSuccess);
00240 
00241     // Success?  If not, clear the pool.
00242     if (!bSuccess)
00243         cAvailThreadPool.clear();
00244     else
00245     { // Success! Mark stored threads as allocated.
00246         iReturn = cAvailThreadPool.size();
00247         for (iIndex = 0; iIndex < iReturn; iIndex++)
00248         {
00249             pThread = cAvailThreadPool[iIndex];
00250             if (pThread)
00251                 pThread->bThreadAllocated = TRUE;
00252         };
00253     };
00254 
00255     // Now it's safe to unlock the critical section
00256     cPoolCritSect.Unlock();
00257 
00258     return (iReturn);
00259 };
