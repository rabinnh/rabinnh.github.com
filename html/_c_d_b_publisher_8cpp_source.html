00081 // Publish records.  This will copy the passed in array and return immediately
00082 
00083 bool CDBPublisher::Publish(CDBRecordsArray &cRecords)
00084 {
00085         // Can't do it if we're still publishing the last call, 10
00086         if (cPublishMutex.Lock(0) == XPLAT_TIMEOUT)
00087                 return false; // Don't wait at all
00088 
00089         // Lock and copy the public array to our private arrayy
00090         vDBRecords = *cRecords.LockDBRecords();
00091         cDBStats.fReqElapsedTotal += cRecords.fTotalElapsed;
00092         cRecords.fTotalElapsed = 0.0f;
00093         // Clear the caller's array
00094         cRecords.ClearDBRecords();
00095         // Unlock the caller
00096         cRecords.UnlockDBRecords();
00097         // We clear the existing buffer before the test so we don't have a memory leak on failure
00098         if (sHost.empty())
00099         {
00100                 cPublishMutex.Unlock();
00101                 return false;
00102         }
00103 
00104         // Start the publishing thread, which will unlock the mutex
00105         cStartEvent.SetEvent();
00106 
00107         return true;
00108 }

00113 unsigned CDBPublisher::WorkerFunction(void *pParam)
00114 {
00115         vector<BSONObj> vBSON;
00116         bool bLocked = true;
00117 
00118         while(true)
00119         { // Wait until we know that we have something to publish
00120                 cStartEvent.Lock();
00121                 tLastPublished = ::time(NULL);
00122                 bool bAuthenticate = false;
00123                 // Do they want us to quit
00124                 if (bTerminate)
00125                 {
00126                         cPublishMutex.Unlock();
00127                         break;
00128                 }
00129                 // Ensure that we can connect and do our thing
00130                 try
00131                 { // Connect and authenticate
00132                         string sErrmsg;
00133                         DBClientConnection cMongoConnect;
00134                         // The catch block will catch a failed connection and Mongo will handle the connection when we go out of scope
00135                         bool bCheckReturn;
00136                         bCheckReturn = cMongoConnect.connect(sHost.c_str(), sErrmsg);
00137                         if (!bCheckReturn)
00138                         {
00139                                 ::openlog("Mongo publisher", LOG_PID, LOG_LOCAL0);
00140                                 ::syslog(LOG_CRIT, "Unable to connect to Mongo database host %s", sDatabase.c_str());
00141                                 ::closelog();
00142                         }
00143                         // Note that the "digest-password" flag has to be "true" for plain text passwords
00144                         if (!sUser.empty() && !sPassword.empty() && !cMongoConnect.auth(sDatabase, sUser, sPassword, sErrmsg, true))
00145                         {
00146                                 cDBStats.lAuthErrors++;
00147                                 cPublishMutex.Unlock();
00148                                 continue;
00149                         }
00150                         bAuthenticate = true;
00151                         int iX;
00152                         string sRemoveArray;
00153                         // Create the BSON records and the array of records that we are going to replace
00154                         // The problem here is that MongoDB has bulk insert, and an update verb, but not a bulk update
00155                         for (iX = 0; iX < vDBRecords.size(); iX++)
00156                         {
00157                                 BSONObj cBSON = fromjson(vDBRecords[iX]);
00158                                 vBSON.push_back(cBSON);
00159                                 // We'll create a JSON array of all the nodes we need to remove so we can do it in one call.
00160                                 if (iX)
00161                                         sRemoveArray += ',';
00162                                 sRemoveArray += vDBRecords[iX];
00163                         }
00164 
00165                         // We get one shot at this.  If it doesn't work we just throw the records away.
00166                         // Ensure that we have the indices
00167                         for (iX = 0; iX < vKeys.size(); iX++)
00168                         {
00169                                 bCheckReturn = cMongoConnect.ensureIndex(sDB_Collection.c_str(), fromjson(vKeys[iX].c_str()), true);
00170                                 if (!bCheckReturn)
00171                                 {
00172                                         ::openlog("Mongo publisher", LOG_PID, LOG_LOCAL0);
00173                                         ::syslog(LOG_CRIT, "ensureIndex failed on database %s, collection %s, index %s\n",
00174                                                 sDatabase.c_str(), sCollection.c_str(), vKeys[iX].c_str());
00175                                         ::closelog();
00176                                 }
00177                         }
00178 
00179                         // Remove the records that we're going to replace.  2 calls is the best that we can do.
00180                         if (bUpsert)
00181                         {
00182                                 // *** RAB - substituion parameter must be replaced with keys
00183                                 string sRemove = pRemoveStart;
00184                                 sRemove += sRemoveArray;
00185                                 sRemove += pRemoveEnd;
00186                                 cMongoConnect.remove(sDB_Collection.c_str(), fromjson(sRemove));
00187                         }
00188 
00189                         // Now insert the records
00190                         cMongoConnect.insert(sDB_Collection.c_str(), vBSON);
00191 
00192                         // Clear our arrays
00193                         vDBRecords.clear();
00194                         vBSON.clear();
00195 
00196                 }
00197                 catch(...)
00198                 {
00199                         bAuthenticate ? cDBStats.lInsertErrors++ : cDBStats.lInsertErrors++;
00200                         if (bLocked)
00201                                 cPublishMutex.Unlock();
00202                         vBSON.clear();
00203                         // If we simply can't reach mongo, we don't want to keep adding records.
00204                         if (vDBRecords.size() > 50000)
00205                                 vDBRecords.clear();
00206                 }
00207                 // Unlock our stuff
00208                 cPublishMutex.Unlock();
00209         }
00210         return 0;
00211 }
